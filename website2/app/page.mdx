<h2 className="fid-section-title">Up and running in three steps</h2>
<p className="fid-section-sub">Install, initialize, and render reactive data.</p>

```bash
flutter pub add flutter_instantdb
```

```dart
// Initialize once, near app start
final db = await InstantDB.init(
  appId: 'your-app-id',
  config: const InstantConfig(syncEnabled: true),
);

// Reactive list — rebuilds automatically as data syncs
InstantBuilder(
  query: {'todos': {}},
  builder: (context, data) {
    final todos = (data['todos'] as List).cast<Map<String, dynamic>>();
    return ListView.builder(
      itemCount: todos.length,
      itemBuilder: (context, i) => Text(todos[i]['title'] as String),
    );
  },
)

// Mutate — optimistic locally, synced to every client
await db.transact([
  ...db.create('todos', {'id': db.id(), 'title': 'Learn InstantDB', 'done': false}),
]);
```
